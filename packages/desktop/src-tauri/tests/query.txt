// PORTABLE GENERATOR RESULTS - verified against parser

// Fetch all nodes of type User
QUERY GetAllUser() =>
    res <- N<User>
    RETURN res


// Fetch User by ID
QUERY GetUserById(target_id: ID) =>
    res <- N<User>(target_id)
    RETURN res

// Traverse Follows from User to User
QUERY TraverseFollows(start_id: ID) =>
    start <- N<User>(start_id)
    res <- start::Out<Follows>
    RETURN res


// Reverse Follows traversal
QUERY InboundFollows(end_id: ID) =>
    end <- N<User>(end_id)
    res <- end::In<Follows>
    RETURN res
